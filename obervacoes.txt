		"SELECT id_user, user_name, nick, email, created_at, upated_at FROM users where user_name LIKE $1 OR nick LIKE $2",
		nameOrNick, nameOrNick,
	)
	if erro != nil {
		return nil, erro
	}
	defer rows.Close()

	var users []entity.UsersEntity
	for rows.Next() {

		if erro = rows.Scan(

			&user.Updated_at,
